JavaScript Usage:

// Alert Modal MirageModal.alert('اطلاعیه', 'عملیات با موفقیت انجام شد!');

// Confirm Modal MirageModal.confirm('تأیید', 'آیا مطمئن هستید؟').then(result => { if (result) console.log('تأیید شد'); });

// Success Modal MirageModal.success('موفقیت', 'اطلاعات ذخیره شد!');

// Error Modal MirageModal.error('خطا', 'مشکلی رخ داده است.');

راهنمای نصب

1. Import CSS and JavaScript:

<!-- CSS --> <link rel="stylesheet" href="components/modal/modal.css">

<!-- JavaScript --> <script src="components/modal/modal.js"></script>

<!-- Feather Icons --> <script src="https://unpkg.com/feather-icons"></script>

2. Basic Usage:

// Create modal const modal = new MirageModal({ size: 'md', variant: 'info' });

// Set content modal.setHeader('عنوان مودال', true); modal.setBody('<p>محتوای مودال</p>'); modal.setFooter([ { text: 'بستن', class: 'mir-btn-secondary', onclick: () => modal.close() } ]);

// Open modal modal.open();
